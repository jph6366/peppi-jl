use peppi::game::{Start, ICE_CLIMBERS};

use peppi::io::slippi::de::Opts as SlippiReadOpts;

fn port_occupancy(start: &Start) -> Vec<PortOccupancy> {
	start
		.players
		.iter()
		.map(|p| PortOccupancy {
			port: p.port,
			follower: p.character == ICE_CLIMBERS,
		})
		.collect()
}

/// Wrapper around peppi's Game structure
#[derive(OpaqueType)]
#[jlrs(key = "SlippiGame")]
pub struct SlippiGame(ImmutableGame);

impl SlippiGame {
    /// Creates a new SlippiGame from a file path
    pub fn from_path(path: JuliaString) -> JlrsResult<CCallRefRet<Self>> {
        let path_str = path.as_str().map_err(|_| JlrsError::exception("Invalid path string"))?;
        
        let game = peppi::io::slippi::read(
            &mut io::BufReader::new(
                fs::File::open(path_str).map_err(|_| JlrsError::exception("Failed to open file"))?
            ),
            None,
        ).map_err(|_| JlrsError::exception("Failed to read Slippi file"))?;
        
        let handle = unsafe { weak_handle_unchecked!() };
        let slippi_game = SlippiGame(game);
        Ok(CCallRefRet::new(TypedValue::new(handle, slippi_game).leak()))
    }
    
    /// Returns the last frame number
    pub fn last_frame(&self) -> i32 {
        self.0.metadata.as_ref()
            .and_then(|m| m.get("lastFrame"))
            .and_then(|v| v.as_i64())
            .unwrap_or(-1) as i32
    }
    
    /// Returns the number of frames in the game
    pub fn frame_count(&self) -> i32 {
        self.0.frames.len() as i32
    }
}

// Helper functions to access game data - these serialize to JSON
// Julia code will call these and parse the JSON on the Julia side
fn get_start_json(game_ref: CCallRef<SlippiGame>) -> JlrsResult<()> {
    let game = game_ref.as_ref()?;
    let json = serde_json::to_string(&game.0.start)
        .map_err(|e| JlrsError::exception(format!("Failed to serialize start: {}", e)))?;
    println!("{}", json);
    Ok(())
}

fn get_end_json(game_ref: CCallRef<SlippiGame>) -> JlrsResult<()> {
    let game = game_ref.as_ref()?;
    let json = serde_json::to_string(&game.0.end)
        .map_err(|e| JlrsError::exception(format!("Failed to serialize end: {}", e)))?;
    println!("{}", json);
    Ok(())
}

fn get_metadata_json(game_ref: CCallRef<SlippiGame>) -> JlrsResult<()> {
    let game = game_ref.as_ref()?;
    let json = serde_json::to_string(&game.0.metadata)
        .map_err(|e| JlrsError::exception(format!("Failed to serialize metadata: {}", e)))?;
    println!("{}", json);
    Ok(())
}

fn get_hash_hex(game_ref: CCallRef<SlippiGame>) -> JlrsResult<()> {
    let game = game_ref.as_ref()?;
    let hex: String = game.0.hash.iter()
        .map(|b| format!("{:02x}", *b))
        .collect();
    println!("{}", hex);
    Ok(())
}


fn read_slippi_path(path: JuliaString) -> JlrsResult<CCallRefRet<SlippiGame>> {
    SlippiGame::from_path(path)
}

fn read_peppi_path(path: JuliaString) -> i32 {
	let path_str = match path.as_str() {
		Ok(s) => s,
		Err(_) => return -1,
	};
	
	let _game = match peppi::io::peppi::read(
		&mut io::BufReader::new(match fs::File::open(path_str) {
			Ok(f) => f,
			Err(_) => return -2,
		}),
		None,
	) {
		Ok(g) => g,
		Err(_) => return -3,
	};
	
	// Successfully read the game
	0
}


julia_module! {
	become julia_app_init_fn;
	
	/// SlippiGame
	///
	/// A wrapper around a parsed Slippi replay file.
	struct SlippiGame;
	
	/// read_slippi_path(path::String)
	///
	/// Read a Slippi replay file from the given path and return a SlippiGame object.
	fn read_slippi_path(path: JuliaString) -> JlrsResult<CCallRefRet<SlippiGame>>;
	
	/// slippi_game_last_frame(game::SlippiGame)
	///
	/// Get the last frame number from the game metadata.
	#[untracked_self]
	in SlippiGame fn last_frame(&self) -> i32 as slippi_game_last_frame;
	
	/// slippi_game_frame_count(game::SlippiGame)
	///
	/// Get the total number of frames in the game.
	#[untracked_self]
	in SlippiGame fn frame_count(&self) -> i32 as slippi_game_frame_count;
	
	/// get_start_json(game::CCallRef{SlippiGame})
	///
	/// Print the start data as JSON (capture with IOBuffer or redirect_stdout).
	fn get_start_json(game_ref: CCallRef<SlippiGame>) -> JlrsResult<()>;
	
	/// get_end_json(game::CCallRef{SlippiGame})
	///
	/// Print the end data as JSON (capture with IOBuffer or redirect_stdout).
	fn get_end_json(game_ref: CCallRef<SlippiGame>) -> JlrsResult<()>;
	
	/// get_metadata_json(game::CCallRef{SlippiGame})
	///
	/// Print the metadata as JSON (capture with IOBuffer or redirect_stdout).
	fn get_metadata_json(game_ref: CCallRef<SlippiGame>) -> JlrsResult<()>;
	
	/// get_hash_hex(game::CCallRef{SlippiGame})
	///
	/// Print the game hash as hex string (capture with IOBuffer or redirect_stdout).
	fn get_hash_hex(game_ref: CCallRef<SlippiGame>) -> JlrsResult<()>;
	
	fn read_peppi_path(path: JuliaString) -> i32;
}
